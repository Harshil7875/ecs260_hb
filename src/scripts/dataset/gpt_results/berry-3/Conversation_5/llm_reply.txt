static void class_inherit(bparser *parser, bexpdesc *e)
{
    if (next_type(parser) == OptColon) { /* ':' */
        
        bexpdesc e1;
        scan_next_token(parser); /* skip ':' */
        expr(parser, &e1);
        check_var(parser, &e1);
        if (e1.k == VNONRELOC) {
            be_code_setsuper(parser->finfo, e, &e1);
        } else {
            berror(parser, "syntax_error: register overflow (more than 255)");
        }
    }
}
